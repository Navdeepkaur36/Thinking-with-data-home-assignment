13. A robust summary function with input checks

# Defining the robust summary function
summarise_cols <- function(df, cols) {
  # Checking for Error with a clear message if any column name is missing
  if (!all(cols %in% names(df))) {
    stop("Error: One or more requested columns are not present in the data frame.")
  }
  # Checking for Error if any requested column is not numeric
  is_numeric_check <- sapply(df[cols], is.numeric)
  if (!all(is_numeric_check)) {
    stop("Error: One or more requested columns are not numeric.")
  }
  # Computing the required metrics using lapply
  summary_list <- lapply(cols, function(col_name) {
    # Extracting the column safely using double brackets
    x <- df[[col_name]]
    # Returning a single-row data frame for this variable
    data.frame(
      n = length(x),
      mean = mean(x, na.rm = TRUE),
      sd = sd(x, na.rm = TRUE),
      na_count = sum(is.na(x)),
      row.names = col_name
    )
  })
  # Combine the list of single-row data frames into one data frame
  result_df <- do.call(rbind, summary_list)
  return(result_df)
}
# Demonstrating the function on mtcars
cat("Summary of mtcars for mpg, hp, and wt:\n")
